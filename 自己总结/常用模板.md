# sort

```cpp
class cmp{
    public:
        bool operator()(const int& a,const int& b){
            return a<b;//从小到大，sort中return true表示符合条件
        }
    };

sort(points.begin(),points.end(),cmp());//放的是cmp类中的比较函数()
sort(points.begin(),points.end(),[](const int& a,const int& b){return a<b;}); 
```

# priority_queue

```cpp
class cmp{
    public:
        bool operator()(const int& a,const int& b){
            return a>b;//从小到大，优先队列中return true表示交换位置。
        }
    };

priority_queue<int,vector<int>,cmp>;//第一个是数据类型，第二个是存放容器，第三个是比较类
```
